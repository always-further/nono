#[cfg(all(test, target_os = "windows"))]
mod broker_job_object_tests {
    use std::path::PathBuf;
    use windows_sys::Win32::Foundation::{BOOL, HANDLE};
    use windows_sys::Win32::System::JobObjects::{IsProcessInJob, JobObjectExtendedLimitInformation, AssignProcessToJobObject, CreateJobObjectW};
    use windows_sys::Win32::System::Threading::{CreateProcessW, ResumeThread, TerminateProcess, CREATE_SUSPENDED, PROCESS_INFORMATION, STARTUPINFOW};

    /// Phase 31 D-04: spawn the broker via CreateProcessW (suspended), assign
    /// to a fresh Job Object, resume, then call `IsProcessInJob(broker_pid, job)`
    /// and assert `in_job != 0`.
    ///
    /// Note: this is a SYNTHETIC test that exercises the Job Object containment
    /// against a controlled spawn, NOT the full BrokerLaunch dispatch. The
    /// dispatch wiring is exercised by Plan 31-05's field-smoke acceptance #1
    /// (real `.\nono.exe shell --profile claude-code` invocation). This unit
    /// test only proves Win32's `AssignProcessToJobObject` call sequence works
    /// for the broker artifact — D-04 expects the broker → child cascade
    /// because Job Object membership is inherited automatically when
    /// `JOB_OBJECT_LIMIT_*BREAKAWAY*` flags are unset.
    #[test]
    fn broker_launch_assigns_child_to_job_object() {
        // Path resolution: same as the production dispatch (current_exe().parent() + broker name).
        // For a unit test, we resolve relative to CARGO_MANIFEST_DIR + target/release.
        let manifest = std::env::var("CARGO_MANIFEST_DIR").expect("CARGO_MANIFEST_DIR set by cargo");
        let workspace_root = std::path::PathBuf::from(&manifest).join("..").join("..");
        let broker_path = workspace_root
            .join("target")
            .join("x86_64-pc-windows-msvc")
            .join("release")
            .join("nono-shell-broker.exe");

        if !broker_path.exists() {
            // Pre-condition: broker built. Plan 31-05 README requires
            // `cargo build -p nono-shell-broker --release --target x86_64-pc-windows-msvc`
            // before running this test. If the artifact isn't present, fail with a clear
            // diagnostic rather than spawning a missing process.
            panic!(
                "broker artifact missing: {} — pre-build via `cargo build -p nono-shell-broker --release --target x86_64-pc-windows-msvc`",
                broker_path.display()
            );
        }

        // Create a fresh Job Object (no resource limits — pure containment test).
        let job: HANDLE = unsafe {
            // SAFETY: CreateJobObjectW with null name and null security attrs is documented to succeed unless OOM.
            CreateJobObjectW(std::ptr::null(), std::ptr::null())
        };
        assert!(!job.is_null(), "CreateJobObjectW failed");

        // Build a noop broker invocation: --shell powershell.exe --shell-arg -Command --shell-arg "exit 0" --cwd <cwd>
        // No --inherit-handle (broker does not need ConPTY for this test; Win32 will spawn without HANDLE_LIST).
        let cwd = std::env::current_dir().expect("cwd").to_string_lossy().into_owned();
        let cmd_line: Vec<u16> = format!(
            r#""{}" --shell C:\Windows\System32\WindowsPowerShell\v1.0\powershell.exe --shell-arg -NoProfile --shell-arg -Command --shell-arg "exit 0" --cwd "{}""#,
            broker_path.display(),
            cwd,
        )
        .encode_utf16()
        .chain(Some(0))
        .collect();
        let mut cmd_buf = cmd_line;

        let mut si: STARTUPINFOW = unsafe { std::mem::zeroed() };
        si.cb = std::mem::size_of::<STARTUPINFOW>() as u32;
        let mut pi: PROCESS_INFORMATION = unsafe { std::mem::zeroed() };

        let created = unsafe {
            // SAFETY: cmd_buf is null-terminated UTF-16; si and pi are zero-initialized.
            CreateProcessW(
                std::ptr::null(),
                cmd_buf.as_mut_ptr(),
                std::ptr::null(),
                std::ptr::null(),
                0,
                CREATE_SUSPENDED,
                std::ptr::null(),
                std::ptr::null(),
                &si,
                &mut pi,
            )
        };
        assert!(created != 0, "CreateProcessW(broker) failed: GetLastError = {}", unsafe { windows_sys::Win32::Foundation::GetLastError() });

        // Assign broker to Job Object BEFORE ResumeThread (D-04 ordering).
        let assigned = unsafe {
            // SAFETY: job and pi.hProcess are valid; AssignProcessToJobObject is documented.
            AssignProcessToJobObject(job, pi.hProcess)
        };
        assert!(assigned != 0, "AssignProcessToJobObject failed: GetLastError = {}", unsafe { windows_sys::Win32::Foundation::GetLastError() });

        // Verify broker is in the Job Object.
        let mut in_job: BOOL = 0;
        let probe = unsafe {
            // SAFETY: in_job is a valid out-pointer; both handles valid.
            IsProcessInJob(pi.hProcess, job, &mut in_job)
        };
        assert!(probe != 0, "IsProcessInJob FFI failed");
        assert!(in_job != 0, "broker process must be in Job Object after AssignProcessToJobObject (D-04)");

        // Resume; broker exits quickly (`exit 0`).
        unsafe {
            // SAFETY: pi.hThread is a valid suspended thread handle.
            ResumeThread(pi.hThread);
        }

        // Cleanup: terminate broker (in case the noop child loop hangs); close handles.
        unsafe {
            // SAFETY: handles are valid for the duration of these calls.
            TerminateProcess(pi.hProcess, 0);
            windows_sys::Win32::Foundation::CloseHandle(pi.hThread);
            windows_sys::Win32::Foundation::CloseHandle(pi.hProcess);
            windows_sys::Win32::Foundation::CloseHandle(job);
        }
    }
}
